P2: verify if a number is prime

VAR x: INTEGER;
READ(x);
VAR i: INTEGER = 2;
VAR prime: INTEGER = 1;
while i*i <= x && prime == 1 DO
BEGIN
	IF x%i == 0) THEN prime = 0;
	i = i+1;
END
IF prime == 1 DO WRITE("Yes, it is prime!")
 ELSE WRITE("No, it is not prime.");
